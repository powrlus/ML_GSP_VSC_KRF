30bool overflowMult(int arg1, int arg2) {
31 // if one factor == 0, no overflow
32 if ((0 == arg1) || (0 == arg2)) {return false;}
33 // Both factors are not equal to 0
34 // Due to the asymetric range of int = -2^31 ... 2^31-1 , two cases must be differed
35 // case 1: result positiv : arg1 * arg2 <= INT_MAX <=> arg1 <= INT_MAX/arg2
36 if (((0 < arg1) && (0 < arg2)) || ((0 > arg1) && (0 > arg2))) {
37 if ((arg1 == INT_MIN) || (arg2 == INT_MIN)){
38 return true; // Since the second factor is not 0, multiplication always returns an overflow.
39 }
40 return (abs(arg1) > (INT_MAX / abs(arg2)));
41 }
42 // one factor is negativ, the other one is positiv
43 int neg_arg = (arg1 > arg2) ? arg2 : arg1;
44 int pos_arg = (arg1 > arg2) ? arg1 : arg2;
45 return (neg_arg < (INT_MIN / pos_arg));
46}
